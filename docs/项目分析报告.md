### 4.6 步骤④：AscendCL 推理集成到 detector.py

**目标**: 在 `YoloOrbbecDetector` 中添加 NPU 推理路径，替代 PyTorch ARM CPU 推理。

#### 4.6.1 验证 NPU 基本推理（先单独测试）

创建 `/home/HwHiAiUser/ultralytics-main/test_npu.py`:

```python
"""
NPU 推理验证脚本
用法: python3 test_npu.py
"""
import numpy as np
import cv2
import time
import os
import sys

# 添加 CANN Python API 路径
sys.path.append('/usr/local/Ascend/ascend-toolkit/latest/python/site-packages')
import acl

class NPUDetector:
    """最简 NPU 检测器——验证 NPU 推理管线是否正常"""

    def __init__(self, om_path, imgsz=640):
        self.imgsz = imgsz
        self.device_id = 0

        # 1. 初始化 ACL
        ret = acl.init()
        assert ret == 0, f"ACL init failed: {ret}"

        # 2. 打开设备
        ret = acl.rt.set_device(self.device_id)
        assert ret == 0, f"Set device failed: {ret}"

        # 3. 加载模型
        self.model_id, _ = acl.mdl.load_from_file(om_path)
        print(f"[NPU] 模型已加载: {om_path}")

        # 4. 获取模型描述
        self.model_desc = acl.mdl.create_desc()
        ret = acl.mdl.get_desc(self.model_desc, self.model_id)
        assert ret == 0

        # 5. 创建输入输出 Dataset
        self.input_dataset, self.output_dataset = acl.mdl.create_dataset()
        # 分配输入 buffer（大小从模型描述获取）
        input_size = acl.mdl.get_input_size_by_index(self.model_desc, 0)
        self.input_buffer, _ = acl.rt.malloc(input_size, acl.ACL_MEM_MALLOC_HUGE_FIRST)
        dataset_buffer = acl.create_data_buffer(self.input_buffer, input_size)
        _, ret = acl.mdl.add_dataset_buffer(self.input_dataset, dataset_buffer)
        assert ret == 0

        # 分配输出 buffer
        output_size = acl.mdl.get_output_size_by_index(self.model_desc, 0)
        self.output_buffer, _ = acl.rt.malloc(output_size, acl.ACL_MEM_MALLOC_HUGE_FIRST)
        out_dataset_buffer = acl.create_data_buffer(self.output_buffer, output_size)
        _, ret = acl.mdl.add_dataset_buffer(self.output_dataset, out_dataset_buffer)
        assert ret == 0

        print(f"[NPU] 输入大小: {input_size}, 输出大小: {output_size}")

    def preprocess(self, image):
        """预处理: BGR→RGB, resize→640×640, HWC→CHW, normalize to [0,1]"""
        img = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
        img = cv2.resize(img, (self.imgsz, self.imgsz))
        img = img.astype(np.float32) / 255.0
        img = np.transpose(img, (2, 0, 1))  # HWC → CHW
        img = np.expand_dims(img, axis=0)    # → (1, 3, 640, 640)
        return np.ascontiguousarray(img)

    def infer(self, image):
        """单帧 NPU 推理"""
        # 预处理
        input_np = self.preprocess(image)

        # 拷贝到 NPU 内存
        acl.rt.memcpy(self.input_buffer, input_np.size * 4,
                      input_np.tobytes(), input_np.size * 4,
                      acl.ACL_MEMCPY_HOST_TO_DEVICE)

        # 执行推理
        ret = acl.mdl.execute(self.model_id, self.input_dataset, self.output_dataset)
        if ret != 0:
            raise RuntimeError(f"Model execute failed: {ret}")

        # 拷贝结果回主机
        output_size = acl.mdl.get_output_size_by_index(self.model_desc, 0)
        output_np = np.zeros(output_size // 4, dtype=np.float32)
        acl.rt.memcpy(output_np.ctypes.data, output_size,
                      self.output_buffer, output_size,
                      acl.ACL_MEMCPY_DEVICE_TO_HOST)

        return output_np

    def close(self):
        acl.rt.free(self.input_buffer)
        acl.rt.free(self.output_buffer)
        acl.mdl.unload(self.model_id)
        acl.mdl.destroy_desc(self.model_desc)
        acl.rt.reset_device(self.device_id)
        acl.finalize()


if __name__ == '__main__':
    print("[NPU 测试] 加载模型...")
    detector = NPUDetector('best4.om', imgsz=640)

    # 测试空推理（预热）
    print("[NPU 测试] 预热推理...")
    dummy = np.random.randint(0, 255, (640, 640, 3), dtype=np.uint8)
    for _ in range(3):
        _ = detector.infer(dummy)

    # 计时测试
    print("[NPU 测试] 计时推理 x100...")
    times = []
    for i in range(100):
        t0 = time.time()
        _ = detector.infer(dummy)
        times.append((time.time() - t0) * 1000)  # ms

    print(f"[NPU 测试] 平均: {np.mean(times):.1f}ms  |  "
          f"最小: {np.min(times):.1f}ms  |  最大: {np.max(times):.1f}ms")
    print(f"[NPU 测试] 预期 NPU 推理: ~15-20ms")
    print(f"[NPU 测试] 对比 CPU: ~120ms (YOLO11n on ARM)")

    detector.close()
    print("[NPU 测试] ✅ 完成")
```

#### 4.6.2 后处理：YOLO 输出解码

NPU 直接输出的是原始张量 `(1, 84, 8400)`（以 YOLO11n 为例：80类+4box × 8400 anchors），需要 NMS 后处理。

创建 `/home/HwHiAiUser/ultralytics-main/npu_postprocess.py`:

```python
"""
NPU 推理后处理 — YOLO 输出解码 + NMS
"""
import numpy as np
import cv2


def nms(boxes, scores, iou_thres=0.45):
    """非极大值抑制（CPU 上做，很快）"""
    indices = cv2.dnn.NMSBoxes(
        boxes.tolist(), scores.tolist(),
        score_threshold=0.5, nms_threshold=iou_thres
    )
    if len(indices) == 0:
        return np.array([])
    return indices.flatten()


def decode_yolo_output(output_np, img_w, img_h, conf_thres=0.5, iou_thres=0.45):
    """
    解码 NPU 输出为检测框列表
    output_np: np.float32 shape (1, 84, 8400) — 以 YOLO11n 为例
    返回: [(x1, y1, x2, y2, conf, cls_id), ...]
    """
    # YOLO11n 输出格式: [batch, 4+nc, anchors]
    output = output_np[0]  # (84, 8400)
    nc = output.shape[0] - 4  # 类别数
    boxes_raw = output[:4, :]  # (4, 8400) — cx, cy, w, h
    scores_raw = output[4:, :]  # (nc, 8400)

    # 每个 anchor 取最大置信度类别
    max_scores = scores_raw.max(axis=0)  # (8400,)
    max_cls = scores_raw.argmax(axis=0)  # (8400,)

    # 置信度过滤
    mask = max_scores > conf_thres
    if not mask.any():
        return np.array([])

    boxes_raw = boxes_raw[:, mask]
    max_scores = max_scores[mask]
    max_cls = max_cls[mask]

    # cx,cy,w,h → x1,y1,x2,y2
    cx, cy, w, h = boxes_raw[0], boxes_raw[1], boxes_raw[2], boxes_raw[3]
    x1 = (cx - w / 2) * img_w / 640
    y1 = (cy - h / 2) * img_h / 640
    x2 = (cx + w / 2) * img_w / 640
    y2 = (cy + h / 2) * img_h / 640

    boxes = np.stack([x1, y1, x2, y2], axis=1)

    # NMS
    keep = nms(boxes, max_scores, iou_thres)
    if len(keep) == 0:
        return np.array([])

    result = np.column_stack([
        boxes[keep],
        max_scores[keep],
        max_cls[keep].astype(np.float32)
    ])
    return result
```

### 5.3 OCR模块详析与优化路线

#### 5.3.1 当前实现（v1 — feat/ocr分支）

**方案**：Tesseract OCR（Google维护的开源引擎，预训练模型，无需自己训练）

**优势**：
- 零训练成本：`apt install` 即用
- ARM原生：C++引擎，香橙派CPU直接运行
- 轻量级：内存占用 < 50MB
- 中英混合：`chi_sim+eng` 语言包覆盖比赛场景

**当前局限**：
- 弯曲表面（易拉罐）识别率下降
- 反光/强阴影下需要更强的预处理
- 关键词匹配是硬匹配，不支持模糊语义（"高等数学上册"匹配不到"数学"的问题不存在，因为用了子串匹配，但顺序敏感）
- Tesseract对非标准字体（艺术字、手写体）效果差

#### 5.3.3 v1.1 多尺度预处理管线（建议赛前实施）

```python
def preprocess_multiscale(self, crop_bgr):
    """多尺度预处理：对同一ROI生成3个预处理变体，OCR后投票"""
    results = []
    # 变体1: CLAHE + OTSU (当前方案，适合平整表面)
    gray = cv2.cvtColor(crop_bgr, cv2.COLOR_BGR2GRAY)
    clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8,8))
    gray1 = clahe.apply(gray)
    _, bin1 = cv2.threshold(gray1, 0, 255, cv2.THRESH_BINARY+cv2.THRESH_OTSU)
    results.append(bin1)

    # 变体2: 自适应阈值 (适合光照不均)
    bin2 = cv2.adaptiveThreshold(gray1, 255,
        cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY, 11, 2)
    results.append(bin2)

    # 变体3: 锐化 + 二值化 (适合模糊文字)
    kernel = np.array([[-1,-1,-1],[-1,9,-1],[-1,-1,-1]])
    sharp = cv2.filter2D(gray, -1, kernel)
    _, bin3 = cv2.threshold(sharp, 0, 255, cv2.THRESH_BINARY+cv2.THRESH_OTSU)
    results.append(bin3)

    return results  # OCR后投票选出最一致的文本
```
